const FOLDER_ID = '17iJ7Wupzh8qm3Ofob8H439rdajHLC4Cw'; const SPREADSHEET_ID = '1tJRmftiJnrdtYLHzoYmD7nnp-4G3GeCtcbNDn5KKUSw'; function escanearDisenos() { try { const ss = SpreadsheetApp.openById(SPREADSHEET_ID); const hoja = ss.getSheets()[0]; const carpetaPrincipal = DriveApp.getFolderById(FOLDER_ID); console.log("Iniciando escaneo reforzado en: " + carpetaPrincipal.getName()); let datos = [['Nombre del Diseño', 'URL Miniatura', 'URL Archivo']]; // Ejecutar búsqueda buscarSoloJPG(carpetaPrincipal, datos); if (datos.length > 1) { hoja.clear(); // Escribimos los datos por bloques para evitar errores de servicio hoja.getRange(1, 1, datos.length, datos[0].length).setValues(datos); SpreadsheetApp.flush(); console.log("¡Éxito! Se procesaron " + (datos.length - 1) + " diseños."); } else { console.warn("No se encontraron imágenes JPG."); } } catch (e) { console.error("Error detectado: " + e.toString()); } } function buscarSoloJPG(carpeta, datos) { try { const archivos = carpeta.getFiles(); while (archivos.hasNext()) { const archivo = archivos.next(); const nombre = archivo.getName(); if (/\.(jpg|jpeg)$/i.test(nombre)) { const id = archivo.getId(); const nombreLimpio = nombre.replace(/\.(jpg|jpeg)$/i, ""); const urlMiniatura = `https://drive.google.com/thumbnail?id=${id}&sz=w500`; const urlArchivo = archivo.getUrl(); datos.push([nombreLimpio, urlMiniatura, urlArchivo]); } } const subcarpetas = carpeta.getFolders(); while (subcarpetas.hasNext()) { // Un pequeño respiro de 50ms para evitar el error de Drive Utilities.sleep(50); buscarSoloJPG(subcarpetas.next(), datos); } } catch (err) { console.warn("Saltando una carpeta por error de acceso o saturación."); } }
