/********************************************************* CONFIGURACIÓN **********************************************************/ const FOLDER_ID = '17iJ7Wupzh8qm3Ofob8H439rdajHLC4Cw'; const SPREADSHEET_ID = '1tJRmftiJnrdtYLHzoYmD7nnp-4G3GeCtcbNDn5KKUSw'; const LIMITE_EJECUCION_MS = 5 * 60 * 1000; const PAGE_SIZE = 500; /********************************************************* FUNCIÓN MAESTRA CON ESTADÍSTICAS **********************************************************/ function escaneoUltraPro() { const inicio = Date.now(); const props = PropertiesService.getScriptProperties(); const estado = JSON.parse(props.getProperty("estado") || "{}"); const ultimaFecha = estado.ultimaFecha || "2000-01-01T00:00:00Z"; let pageToken = estado.pageToken || null; const ss = SpreadsheetApp.openById(SPREADSHEET_ID); const hoja = ss.getSheets()[0]; const ultimaFila = hoja.getLastRow(); let nombresExistentes = new Set(); if (ultimaFila > 1) { const datos = hoja.getRange(2, 1, ultimaFila - 1, 1).getValues(); nombresExistentes = new Set(datos.flat()); } const nuevosDatos = []; // 🔢 MÉTRICAS let totalAnalizados = 0; let totalNuevos = 0; let totalDuplicados = 0; do { if (Date.now() - inicio > LIMITE_EJECUCION_MS) { guardarEstado(ultimaFecha, pageToken); crearTriggerContinuacion(); console.log("⏳ Se continuará automáticamente..."); mostrarResumen(totalAnalizados, totalNuevos, totalDuplicados, hoja); return; } const response = Drive.Files.list({ q: `'${FOLDER_ID}' in parents and mimeType='image/jpeg' and modifiedTime > '${ultimaFecha}' and trashed=false`, fields: "nextPageToken, files(id, name, modifiedTime)", pageSize: PAGE_SIZE, pageToken: pageToken }); const archivos = response.files || []; archivos.forEach(archivo => { totalAnalizados++; const nombreLimpio = archivo.name.replace(/\.(jpg|jpeg)$/i, ""); if (!nombresExistentes.has(nombreLimpio)) { nuevosDatos.push([ nombreLimpio, // 🔥 Miniatura optimizada para carga rápida `https://drive.google.com/thumbnail?id=${archivo.id}&sz=w150`, `https://drive.google.com/file/d/${archivo.id}/view` ]); nombresExistentes.add(nombreLimpio); totalNuevos++; } else { totalDuplicados++; } }); pageToken = response.nextPageToken; } while (pageToken); if (nuevosDatos.length > 0) { hoja.getRange(hoja.getLastRow() + 1, 1, nuevosDatos.length, 3) .setValues(nuevosDatos); } props.setProperty("estado", JSON.stringify({ ultimaFecha: new Date().toISOString(), pageToken: null })); eliminarTriggerContinuacion(); mostrarResumen(totalAnalizados, totalNuevos, totalDuplicados, hoja); } /********************************************************* RESUMEN FINAL **********************************************************/ function mostrarResumen(totalAnalizados, totalNuevos, totalDuplicados, hoja) { const totalEnHoja = hoja.getLastRow() - 1; console.log("================================="); console.log("📊 RESUMEN DE ESCANEO"); console.log("Archivos analizados: " + totalAnalizados); console.log("Nuevos agregados: " + totalNuevos); console.log("Duplicados omitidos: " + totalDuplicados); console.log("Total acumulado en hoja: " + totalEnHoja); console.log("================================="); } /********************************************************* GESTIÓN DE ESTADO **********************************************************/ function guardarEstado(ultimaFecha, pageToken) { PropertiesService.getScriptProperties().setProperty( "estado", JSON.stringify({ ultimaFecha, pageToken }) ); } function crearTriggerContinuacion() { ScriptApp.newTrigger("escaneoUltraPro") .timeBased() .after(1 * 60 * 1000) .create(); } function eliminarTriggerContinuacion() { const triggers = ScriptApp.getProjectTriggers(); triggers.forEach(t => { if (t.getHandlerFunction() === "escaneoUltraPro") { ScriptApp.deleteTrigger(t); } }); }
