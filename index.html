const FOLDER_ID = '17iJ7Wupzh8qm3Ofob8H439rdajHLC4Cw'; const SPREADSHEET_ID = '1tJRmftiJnrdtYLHzoYmD7nnp-4G3GeCtcbNDn5KKUSw'; function escanearDisenos() { try { const ss = SpreadsheetApp.openById(SPREADSHEET_ID); const hoja = ss.getSheets()[0]; const carpetaPrincipal = DriveApp.getFolderById(FOLDER_ID); console.log("Iniciando escaneo EXCLUSIVO de JPG en: " + carpetaPrincipal.getName()); // Encabezados let datos = [['Nombre del Diseño', 'URL Miniatura', 'URL Archivo']]; // Iniciar búsqueda recursiva buscarSoloJPG(carpetaPrincipal, datos); hoja.clear(); if (datos.length > 1) { hoja.getRange(1, 1, datos.length, datos[0].length).setValues(datos); SpreadsheetApp.flush(); console.log("¡Hecho! Se encontraron " + (datos.length - 1) + " archivos JPG."); } else { console.warn("⚠️ No se encontró ningún archivo con extensión .jpg o .jpeg."); } } catch (e) { console.error("❌ Error: " + e.toString()); } } function buscarSoloJPG(carpeta, datos) { const archivos = carpeta.getFiles(); while (archivos.hasNext()) { const archivo = archivos.next(); const nombreOriginal = archivo.getName(); // FILTRO ULTRA ESTRICTO: Solo .jpg o .jpeg (ignora mayúsculas/minúsculas) if (/\.(jpg|jpeg)$/i.test(nombreOriginal)) { const id = archivo.getId(); // Quitamos la extensión del nombre para que se vea limpio en la web const nombreLimpio = nombreOriginal.replace(/\.(jpg|jpeg)$/i, ""); const urlMiniatura = `https://drive.google.com/thumbnail?id=${id}&sz=w500`; const urlArchivo = archivo.getUrl(); datos.push([nombreLimpio, urlMiniatura, urlArchivo]); console.log("JPG Agregado: " + nombreLimpio); } else { console.log("Ignorado (no es JPG): " + nombreOriginal); } } // Entrar en subcarpetas const subcarpetas = carpeta.getFolders(); while (subcarpetas.hasNext()) { buscarSoloJPG(subcarpetas.next(), datos); } }
